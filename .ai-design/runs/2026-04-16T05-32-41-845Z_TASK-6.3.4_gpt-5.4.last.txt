+    [Fact]
+    public async Task Internal_finance_write_endpoints_mutate_records_for_authorized_tenant()
+    {
+        var seed = await SeedFinanceWriteCompanyAsync();
+        using var client = CreateAuthenticatedClient(seed.Subject, seed.Email, seed.DisplayName);
+
+        var invoiceResponse = await client.PatchAsJsonAsync(
+            $"/internal/companies/{seed.CompanyId}/finance/invoices/{seed.InvoiceId}/approval-status",
+            new { Status = "approved" });
+        var transactionResponse = await client.PatchAsJsonAsync(
+            $"/internal/companies/{seed.CompanyId}/finance/transactions/{seed.TransactionId}/category",
+            new { Category = "software_subscriptions" });
+
+        Assert.Equal(HttpStatusCode.OK, invoiceResponse.StatusCode);
+        Assert.Equal(HttpStatusCode.OK, transactionResponse.StatusCode);
+
+        var invoice = await invoiceResponse.Content.ReadFromJsonAsync<InvoiceResponse>();
+        var transaction = await transactionResponse.Content.ReadFromJsonAsync<TransactionResponse>();
+        Assert.NotNull(invoice);
+        Assert.NotNull(transaction);
+        Assert.Equal("approved", invoice!.Status);
+        Assert.Equal("software_subscriptions", transaction!.TransactionType);
+
+        using var scope = _factory.Services.CreateScope();
+        var dbContext = scope.ServiceProvider.GetRequiredService<VirtualCompanyDbContext>();
+        Assert.Equal(
+            "approved",
+            await dbContext.FinanceInvoices
+                .IgnoreQueryFilters()
+                .Where(x => x.Id == seed.InvoiceId)
+                .Select(x => x.Status)
+                .SingleAsync());
+        Assert.Equal(
+            "software_subscriptions",
+            await dbContext.FinanceTransactions
+                .IgnoreQueryFilters()
+                .Where(x => x.Id == seed.TransactionId)
+                .Select(x => x.TransactionType)
+                .SingleAsync());
+    }

+    [Fact]
+    public async Task Finance_policy_configuration_is_retrievable_through_endpoint_and_typed_service()
+    {
+        var seed = await SeedFinanceWriteCompanyAsync();
+        using var client = CreateAuthenticatedClient(seed.Subject, seed.Email, seed.DisplayName);
+        var configuration = new FinancePolicyConfigurationDto(
+            seed.CompanyId,
+            "eur",
+            12500m,
+            7500m,
+            true,
+            -2500m,
+            15000m,
+            120,
+            45);
+
+        var saveResponse = await client.PutAsJsonAsync(
+            $"/internal/companies/{seed.CompanyId}/finance/policy-configuration",
+            configuration);
+        var getResponse = await client.GetAsync(
+            $"/internal/companies/{seed.CompanyId}/finance/policy-configuration");
+
+        Assert.Equal(HttpStatusCode.OK, saveResponse.StatusCode);
+        Assert.Equal(HttpStatusCode.OK, getResponse.StatusCode);
+
+        var saved = await saveResponse.Content.ReadFromJsonAsync<PolicyConfigurationResponse>();
+        var retrieved = await getResponse.Content.ReadFromJsonAsync<PolicyConfigurationResponse>();
+        Assert.NotNull(saved);
+        Assert.NotNull(retrieved);
+        Assert.Equal(seed.CompanyId, retrieved!.CompanyId);
+        Assert.Equal("EUR", retrieved.ApprovalCurrency);
+        Assert.Equal(12500m, retrieved.InvoiceApprovalThreshold);
+        Assert.Equal(7500m, retrieved.BillApprovalThreshold);
+        Assert.Equal(-2500m, retrieved.AnomalyDetectionLowerBound);
+        Assert.Equal(15000m, retrieved.AnomalyDetectionUpperBound);
+        Assert.Equal(120, retrieved.CashRunwayWarningThresholdDays);
+        Assert.Equal(45, retrieved.CashRunwayCriticalThresholdDays);
+        Assert.Equal(saved, retrieved);
+
+        using var scope = _factory.Services.CreateScope();
+        scope.ServiceProvider.GetRequiredService<ICompanyContextAccessor>().SetCompanyId(seed.CompanyId);
+        var policyService = scope.ServiceProvider.GetRequiredService<IFinancePolicyConfigurationService>();
+        var typedConfiguration = await policyService.GetPolicyConfigurationAsync(
+            new GetFinancePolicyConfigurationQuery(seed.CompanyId),
+            CancellationToken.None);
+
+        Assert.Equal("EUR", typedConfiguration.ApprovalCurrency);
+        Assert.Equal(retrieved.InvoiceApprovalThreshold, typedConfiguration.InvoiceApprovalThreshold);
+        Assert.Equal(retrieved.AnomalyDetectionLowerBound, typedConfiguration.AnomalyDetectionLowerBound);
+        Assert.Equal(retrieved.CashRunwayCriticalThresholdDays, typedConfiguration.CashRunwayCriticalThresholdDays);
+    }

+    private async Task<FinanceWriteEndpointSeed> SeedFinanceWriteCompanyAsync()
+    {
+        var userId = Guid.NewGuid();
+        var companyId = Guid.NewGuid();
+        var otherCompanyId = Guid.NewGuid();
+        var subject = $"finance-writer-{Guid.NewGuid():N}";
+        var email = $"{subject}@example.com";
+        const string displayName = "Finance Writer";
+
+        var invoiceId = Guid.NewGuid();
+        var transactionId = Guid.NewGuid();
+        var otherInvoiceId = Guid.NewGuid();
+        var otherTransactionId = Guid.NewGuid();
+
+        await _factory.SeedAsync(dbContext =>
+        {
+            dbContext.Users.Add(new User(userId, email, displayName, "dev-header", subject));
+            dbContext.Companies.AddRange(
+                new Company(companyId, "Finance Write Company"),
+                new Company(otherCompanyId, "Other Finance Write Company"));
+            dbContext.CompanyMemberships.Add(new CompanyMembership(
+                Guid.NewGuid(),
+                companyId,
+                userId,
+                CompanyMembershipRole.Owner,
+                CompanyMembershipStatus.Active));
+
+            SeedFinanceRecords(dbContext, companyId, invoiceId, transactionId);
+            SeedFinanceRecords(dbContext, otherCompanyId, otherInvoiceId, otherTransactionId);
+            return Task.CompletedTask;
+        });
+
+        return new FinanceWriteEndpointSeed(
+            companyId,
+            otherCompanyId,
+            invoiceId,
+            transactionId,
+            otherInvoiceId,
+            otherTransactionId,
+            subject,
+            email,
+            displayName);
+    }
+
+    private static void SeedFinanceRecords(
+        VirtualCompanyDbContext dbContext,
+        Guid companyId,
+        Guid invoiceId,
+        Guid transactionId)
+    {
+        var accountId = Guid.NewGuid();
+        var counterpartyId = Guid.NewGuid();
+
+        dbContext.FinanceAccounts.Add(new FinanceAccount(
+            accountId,
+            companyId,
+            "1000",
+            "Operating Cash",
+            "asset",
+            "USD",
+            1000m,
+            new DateTime(2026, 1, 1, 0, 0, 0, DateTimeKind.Utc)));
+        dbContext.FinanceCounterparties.Add(new FinanceCounterparty(
+            counterpartyId,
+            companyId,
+            "Customer",
+            "customer",
+            "customer@example.com"));
+        dbContext.FinanceInvoices.Add(new FinanceInvoice(
+            invoiceId,
+            companyId,
+            counterpartyId,
+            $"INV-{invoiceId:N}"[..16],
+            new DateTime(2026, 1, 2, 0, 0, 0, DateTimeKind.Utc),
+            new DateTime(2026, 2, 2, 0, 0, 0, DateTimeKind.Utc),
+            2000m,
+            "USD",
+            "open"));
+        dbContext.FinanceTransactions.Add(new FinanceTransaction(
+            transactionId,
+            companyId,
+            accountId,
+            counterpartyId,
+            invoiceId,
+            null,
+            new DateTime(2026, 1, 3, 0, 0, 0, DateTimeKind.Utc),
+            "customer_payment",
+            2000m,
+            "USD",
+            "Customer payment",
+            $"PAY-{transactionId:N}"[..20]));
+    }

+    private HttpClient CreateAuthenticatedClient(string subject, string email, string displayName)
+    {
+        var client = _factory.CreateClient();
+        client.DefaultRequestHeaders.Add(DevHeaderAuthenticationDefaults.SubjectHeader, subject);
+        client.DefaultRequestHeaders.Add(DevHeaderAuthenticationDefaults.EmailHeader, email);
+        client.DefaultRequestHeaders.Add(DevHeaderAuthenticationDefaults.DisplayNameHeader, displayName);
+        return client;
+    }
