+    {
+        var prefix = Prefix();

+    public static IReadOnlyList<string> NormalizeDepartments(IEnumerable<string>? departments) =>
+        (departments ?? [])

+            .Select(x => x.Trim().ToLowerInvariant())
+            .Distinct(StringComparer.OrdinalIgnoreCase)
+            .OrderBy(x => x, StringComparer.OrdinalIgnoreCase)

+        var stopwatch = Stopwatch.StartNew();
+
+        try
+        {
+            var result = await ApiClient.GetWidgetAsync<TPayload>(CompanyId, widgetKey);
+            ApplyWidgetPayload(widgetKey, result?.Payload);
+            Logger.LogInformation(
+                "Executive cockpit widget {WidgetKey} rendered in {ElapsedMilliseconds} ms.",
+                widgetKey,
+                stopwatch.Elapsed.TotalMilliseconds);
+        }
+        catch (OnboardingApiException ex)
+        {
+            widgetErrors[widgetKey] = ex.Message;
+        }
+        finally
+        {
+            loadingWidgets.Remove(widgetKey);
+        }
+    }
+
+    private void ApplyWidgetPayload<TPayload>(string widgetKey, TPayload? payload)
+    {
+        if (dashboard is null || payload is null)
+        {
+            return;
+        }
+
+        switch (widgetKey)
+        {
+            case "summary-kpis" when payload is List<ExecutiveCockpitSummaryKpiViewModel> summaryKpis:
+                dashboard.SummaryKpis = summaryKpis;
+                break;
+            case "daily-briefing" when payload is ExecutiveCockpitDailyBriefingViewModel dailyBriefing:
+                dashboard.DailyBriefing = dailyBriefing;
+                break;
+            case "pending-approvals" when payload is ExecutiveCockpitPendingApprovalsViewModel approvals:
+                dashboard.PendingApprovals = approvals;
+                dashboard.EmptyStateFlags.NoPendingApprovals = approvals.TotalCount == 0;
+                break;
+            case "alerts" when payload is List<ExecutiveCockpitAlertViewModel> alerts:
+                dashboard.Alerts = alerts;
+                dashboard.EmptyStateFlags.NoAlerts = alerts.Count == 0;
+                break;
+        }
+    }
