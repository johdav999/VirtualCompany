+        var invoices = Enumerable.Range(1, 5)

+                $"INV-{now:yyyyMM}-{index:000}",
+                now.AddDays(-45 + index),

+        var bills = Enumerable.Range(1, 5)

+                Guid.NewGuid(),
+                companyId,
+                counterparties[1 + (index % 3)].Id,
+                $"BILL-{now:yyyyMM}-{index:000}",
+                now.AddDays(-40 + index),
+                now.AddDays(-10 + index),
+                900m + (index * 210m),
+                "USD",
+                index % 2 == 0 ? "paid" : "open"))
+            .ToArray();
+
+        var transactions = new List<FinanceTransaction>(capacity: 50);
+        for (var index = 0; index < 50; index++)

+            var isRevenue = index % 2 == 0;
