+    private readonly IFinanceSeedingStateService _financeSeedingStateService;
+    private readonly IBackgroundExecutionIdentityFactory _identityFactory;
+    private readonly IAuditEventWriter _auditEventWriter;
+    private readonly ICorrelationContextAccessor? _correlationContextAccessor;
+    private readonly TimeProvider _timeProvider;
+    private readonly ILogger<CompanyFinanceEntryService> _logger;
+
+    public CompanyFinanceEntryService(
+        VirtualCompanyDbContext dbContext,
+        IFinanceSeedingStateService financeSeedingStateService,
+        IBackgroundExecutionIdentityFactory identityFactory,
+        IAuditEventWriter auditEventWriter,
+        TimeProvider timeProvider,
+        ILogger<CompanyFinanceEntryService> logger,
+        ICorrelationContextAccessor? correlationContextAccessor = null)
+    {
+        _dbContext = dbContext;
+        _financeSeedingStateService = financeSeedingStateService;
+        _identityFactory = identityFactory;
+        _auditEventWriter = auditEventWriter;

+    public async Task<FinanceEntryStateDto> GetEntryStateAsync(GetFinanceEntryStateQuery query, CancellationToken cancellationToken)
+    {
+        if (query.CompanyId == Guid.Empty)
+        {
+            throw new ArgumentException("Company id is required.", nameof(query));
+        }
+
+        var checkedAtUtc = _timeProvider.GetUtcNow().UtcDateTime;
+        var company = await _dbContext.Companies
+            .IgnoreQueryFilters()
+            .SingleOrDefaultAsync(x => x.Id == query.CompanyId, cancellationToken)
+            ?? throw new KeyNotFoundException($"Company '{query.CompanyId}' was not found.");
+
+        var seedingState = await _financeSeedingStateService.GetCompanyFinanceSeedingStateAsync(query.CompanyId, cancellationToken);
+        var execution = await FindExecutionAsync(query.CompanyId, cancellationToken);
+        var seedJobEnqueued = false;
+
+        if (ShouldRetry(query, seedingState.State, execution))
+        {
+            execution!.Queue(
+                checkedAtUtc,
+                _identityFactory.EnsureCorrelationId(_correlationContextAccessor?.CorrelationId),
+                resetAttempts: true);
+            seedJobEnqueued = true;
+            await WriteRequestedAuditAsync(company.Id, execution, query.Source, isRetry: true, cancellationToken);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+        }
+        else if (ShouldEnqueueInitialRequest(seedingState.State, execution))
+        {
+            execution = await EnqueueAsync(company.Id, query.Source, checkedAtUtc, cancellationToken);
+            seedJobEnqueued = true;
+        }
+
+        var initializationStatus = ResolveInitializationStatus(seedingState.State, execution);
+        var canRetry = initializationStatus == FinanceEntryInitializationStates.Failed;
+        var canRefresh = initializationStatus != FinanceEntryInitializationStates.Ready;
+
+        return new FinanceEntryStateDto(
+            company.Id,
+            initializationStatus,
+            seedingState.State,
+            seedJobEnqueued,
+            IsExecutionActive(execution),
+            canRetry,
+            canRefresh,
+            BuildMessage(initializationStatus, execution),
+            checkedAtUtc,
+            company.FinanceSeededUtc,
+            execution?.StartedUtc,
+            execution?.CompletedUtc,
+            execution?.FailureCode,
+            execution?.FailureMessage,
+            execution?.Status.ToStorageValue(),
+            execution?.CorrelationId);
+    }
+
+    private async Task<BackgroundExecution?> EnqueueAsync(
+        Guid companyId,
+        string source,
+        DateTime checkedAtUtc,
+        CancellationToken cancellationToken)
+    {
+        var identity = CreateIdentity(companyId);
+        var execution = new BackgroundExecution(
+            Guid.NewGuid(),
+            companyId,
+            BackgroundExecutionType.FinanceSeed,
+            BackgroundExecutionRelatedEntityTypes.FinanceSeed,
+            companyId.ToString("D"),
+            identity.CorrelationId,
+            identity.IdempotencyKey,
+            maxAttempts: 5);
+
+        _dbContext.BackgroundExecutions.Add(execution);
+
+        try
+        {
+            await WriteRequestedAuditAsync(companyId, execution, source, isRetry: false, cancellationToken);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+            _logger.LogInformation(
+                "Queued finance seed background execution {ExecutionId} for company {CompanyId}.",
+                execution.Id,
+                companyId);
+            return execution;
+        }
+        catch (DbUpdateException)
+        {
+            _dbContext.Entry(execution).State = EntityState.Detached;
+            return await FindExecutionAsync(companyId, cancellationToken);
+        }
+    }
+
+    private async Task WriteRequestedAuditAsync(
+        Guid companyId,
+        BackgroundExecution execution,
+        string source,
+        bool isRetry,
+        CancellationToken cancellationToken)
+    {
+        await _auditEventWriter.WriteAsync(
+            new AuditEventWriteRequest(
+                companyId,
+                AuditActorTypes.System,
+                null,
+                FinanceSeedRequestedAction,
+                BackgroundExecutionRelatedEntityTypes.FinanceSeed,
+                execution.Id.ToString("D"),
+                AuditEventOutcomes.Requested,
+                isRetry
+                    ? "Finance entry requested a retry for the finance seed background job."
+                    : "Finance entry requested the finance seed background job.",
+                Metadata: new Dictionary<string, string?>
+                {
+                    ["source"] = source,
+                    ["companyId"] = companyId.ToString("D"),
+                    ["executionId"] = execution.Id.ToString("D"),
+                    ["jobStatus"] = execution.Status.ToStorageValue(),
+                    ["idempotencyKey"] = execution.IdempotencyKey,
+                    ["correlationId"] = execution.CorrelationId,
+                    ["isRetry"] = isRetry ? "true" : "false"
+                },
+                CorrelationId: execution.CorrelationId,
+                OccurredUtc: _timeProvider.GetUtcNow().UtcDateTime),
+            cancellationToken);
+    }
+
+    private BackgroundExecutionIdentity CreateIdentity(Guid companyId) =>
+        _identityFactory.Create(
+            companyId,
+            "finance-seed",
+            _correlationContextAccessor?.CorrelationId,
+            BackgroundExecutionType.FinanceSeed.ToStorageValue(),
+            companyId.ToString("N"));
+
+    private async Task<BackgroundExecution?> FindExecutionAsync(Guid companyId, CancellationToken cancellationToken)
+    {
+        var identity = _identityFactory.CreateIdempotencyKey(
+            "finance-seed",
+            BackgroundExecutionType.FinanceSeed.ToStorageValue(),
+            companyId.ToString("N"));
+
+        return await _dbContext.BackgroundExecutions
+            .IgnoreQueryFilters()
+            .SingleOrDefaultAsync(
+                x => x.CompanyId == companyId &&
+                     x.ExecutionType == BackgroundExecutionType.FinanceSeed &&
+                     x.IdempotencyKey == identity,
+                cancellationToken);
+    }
+
+    private static bool ShouldEnqueueInitialRequest(FinanceSeedingState seedingState, BackgroundExecution? execution) =>
+        seedingState == FinanceSeedingState.NotSeeded &&
+        execution is null;
+
+    private static bool ShouldRetry(GetFinanceEntryStateQuery query, FinanceSeedingState seedingState, BackgroundExecution? execution) =>
+        query.RetryOnFailure &&
+        seedingState != FinanceSeedingState.FullySeeded &&
+        execution is not null &&
+        execution.Status is BackgroundExecutionStatus.Failed or BackgroundExecutionStatus.Blocked;
+
+    private static string ResolveInitializationStatus(FinanceSeedingState seedingState, BackgroundExecution? execution)
+    {
+        if (seedingState == FinanceSeedingState.FullySeeded)
+        {
+            return FinanceEntryInitializationStates.Ready;
+        }
+
+        if (execution is null)
+        {
+            return FinanceEntryInitializationStates.Initializing;
+        }
+
+        return execution.Status switch
+        {
+            BackgroundExecutionStatus.Pending or
+            BackgroundExecutionStatus.InProgress or
+            BackgroundExecutionStatus.RetryScheduled => FinanceEntryInitializationStates.Initializing,
+            BackgroundExecutionStatus.Failed or
+            BackgroundExecutionStatus.Blocked => FinanceEntryInitializationStates.Failed,
+            _ => FinanceEntryInitializationStates.Ready
+        };
+    }
+
+    private static bool IsExecutionActive(BackgroundExecution? execution) =>
+        execution?.Status is BackgroundExecutionStatus.Pending
+            or BackgroundExecutionStatus.InProgress
+            or BackgroundExecutionStatus.RetryScheduled;
+
+    private static string BuildMessage(string initializationStatus, BackgroundExecution? execution) =>
+        initializationStatus switch
+        {
+            FinanceEntryInitializationStates.Ready =>
+                "Finance data is ready.",
+            FinanceEntryInitializationStates.Failed =>
+                string.IsNullOrWhiteSpace(execution?.FailureMessage)
+                    ? "Finance setup failed. Retry the initialization to prepare the finance workspace."
+                    : $"Finance setup failed: {execution!.FailureMessage}",
+            _ when execution?.Status == BackgroundExecutionStatus.RetryScheduled =>
+                "Finance data is still being prepared. A previous attempt failed and the job is retrying in the background.",
+            _ when execution?.Status == BackgroundExecutionStatus.InProgress =>
+                "Finance data is being prepared in the background. This page will refresh automatically when setup completes.",
+            _ =>
+                "Finance data is being prepared in the background. This page will refresh automatically when setup completes."
+        };
+}

+                    handled++;
+                    break;
+                case BackgroundJobExecutionOutcome.RetryScheduled:
+                    execution.ScheduleRetry(
+                        nowUtc.Add(result.RetryDelay ?? TimeSpan.Zero),
+                        MapFailureCategory(result.FailureClassification),
+                        ResolveFailureCode(result),
+                        ResolveFailureMessage(result));
+                    break;
+                case BackgroundJobExecutionOutcome.Blocked:
+                case BackgroundJobExecutionOutcome.PermanentFailure:
+                case BackgroundJobExecutionOutcome.RetryExhausted:
+                    execution.MarkFailed(
+                        MapFailureCategory(result.FailureClassification),
+                        ResolveFailureCode(result),
+                        ResolveFailureMessage(result));
+                    await WriteLifecycleAuditAsync(
+                        execution,
+                        FinanceSeedFailedAction,
+                        AuditEventOutcomes.Failed,
+                        ResolveFailureMessage(result),
+                        cancellationToken);
+                    handled++;
+                    break;
+            }

+            await _dbContext.SaveChangesAsync(cancellationToken);
+        }

+        return handled;

+                 (x.Status == BackgroundExecutionStatus.InProgress &&
+                  x.HeartbeatUtc != null &&
+                  x.HeartbeatUtc <= staleBeforeUtc)))

+    private static int ResolveSeedValue(Guid companyId)
+    {
+        var bytes = companyId.ToByteArray();
+        var value =
+            BitConverter.ToInt32(bytes, 0) ^
+            BitConverter.ToInt32(bytes, 4) ^
+            BitConverter.ToInt32(bytes, 8) ^
+            BitConverter.ToInt32(bytes, 12);
+
+        if (value == int.MinValue)
+        {
+            return 913;
+        }
+
+        value = Math.Abs(value);
+        return value == 0 ? 913 : value;
+    }

+    private readonly IOptions<FinanceSeedWorkerOptions> _options;

+    public FinanceSeedBackgroundService(
+        IServiceScopeFactory scopeFactory,
+        IOptions<FinanceSeedWorkerOptions> options,
+        ILogger<FinanceSeedBackgroundService> logger)
+    {
+        _scopeFactory = scopeFactory;
+        _options = options;
+        _logger = logger;
+    }
+
+    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
+    {
+        if (!_options.Value.Enabled)
+        {
+            _logger.LogInformation("Finance seed worker is disabled.");
+            return;
+        }
+
+        var pollInterval = TimeSpan.FromMilliseconds(Math.Max(100, _options.Value.PollIntervalMilliseconds));
+        while (!stoppingToken.IsCancellationRequested)
+        {
+            try
+            {
+                using var scope = _scopeFactory.CreateScope();
+                var runner = scope.ServiceProvider.GetRequiredService<IFinanceSeedJobRunner>();
+                var handled = await runner.RunDueAsync(stoppingToken);
+                if (handled == 0)
+                {
+                    await Task.Delay(pollInterval, stoppingToken);
+                }
+            }
+            catch (OperationCanceledException) when (stoppingToken.IsCancellationRequested)
+            {
+                break;
+            }
+            catch (Exception ex)
+            {
+                _logger.LogError(ex, "Finance seed worker loop failed.");
+                await Task.Delay(pollInterval, stoppingToken);
+            }
+        }
+    }
