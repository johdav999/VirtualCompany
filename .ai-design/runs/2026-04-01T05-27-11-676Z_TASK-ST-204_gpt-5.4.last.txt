+    private static AgentWorkloadSummaryDto BuildWorkloadSummary(
+        Agent agent,
+        AgentExecutionAggregate? execution,
+        AgentApprovalAggregate? approval,
+        DateTime? auditActivityUtc)
+    {
+        var awaitingApprovalCount = (execution?.AwaitingApprovalCount ?? 0) + (approval?.PendingApprovalCount ?? 0);
+        var failedCount = execution?.FailedCount ?? 0;
+        var executedCount = execution?.ExecutedCount ?? 0;
+        var openItemsCount = awaitingApprovalCount + failedCount;
+        var lastActivityUtc = MaxUtc(agent.UpdatedUtc, execution?.LastActivityUtc, approval?.LastActivityUtc, auditActivityUtc);
+        var staleThresholdUtc = DateTime.UtcNow.AddDays(-ActivityStaleAfterDays);
+
+        var healthStatus =
+            agent.Status == AgentStatus.Archived ? "archived" :
+            agent.Status is AgentStatus.Paused or AgentStatus.Restricted ? "attention" :
+            lastActivityUtc is null || lastActivityUtc < staleThresholdUtc || openItemsCount >= 3 ? "attention" :
+            "healthy";
+
+        var summary =
+            lastActivityUtc is null
+                ? "No recent activity has been captured yet."
+                : $"{openItemsCount} open items, {awaitingApprovalCount} awaiting approval. Last activity {lastActivityUtc.Value:yyyy-MM-dd HH:mm} UTC.";
+
+        return new AgentWorkloadSummaryDto(
+            openItemsCount,
+            awaitingApprovalCount,
+            executedCount,
+            failedCount,
+            lastActivityUtc,
+            healthStatus,
+            summary);
+    }

+    private static DateTime? MaxUtc(params DateTime?[] values)
+    {
+        DateTime? current = null;
+
+        foreach (var value in values)
+        {
+            if (value is null)
+            {
+                continue;
+            }
+
+            current = current is null || value > current ? value : current;
+        }
+
+        return current;
+    }
