+using System.Globalization;
+
+namespace VirtualCompany.Application.Finance;
+
+public static class FinanceIntelligenceHeuristics
+{
+    public static FinanceIntelligenceSnapshotDto Evaluate(FinanceIntelligenceInputDto input)
+    {
+        ArgumentNullException.ThrowIfNull(input);
+
+        var sevenDayProjection = BuildProjection(input, 7);
+        var thirtyDayProjection = BuildProjection(input, 30);
+        var overdueInvoices = BuildOverdueInvoices(input);
+        var dueSoonBills = BuildDueSoonBills(input, sevenDayProjection);
+        var obligationCoverage = BuildObligationCoverage(input, sevenDayProjection);
+
+        return new FinanceIntelligenceSnapshotDto(
+            input.AsOfUtc,
+            sevenDayProjection,
+            thirtyDayProjection,
+            obligationCoverage,
+            overdueInvoices,
+            dueSoonBills);
+    }
+
+    private static FinanceCashProjectionDto BuildProjection(FinanceIntelligenceInputDto input, int horizonDays)
+    {
+        var horizonEndDate = input.AsOfUtc.Date.AddDays(horizonDays);
+
+        // The projection is intentionally deterministic: it only considers currently open
+        // receivables, bills, and scheduled recurring outflows that are due inside the horizon.
+        var projectedInflows = Round(input.OpenInvoices
+            .Where(x => x.OutstandingAmount > 0m && x.DueUtc.Date <= horizonEndDate)
+            .Sum(x => x.OutstandingAmount));
+
+        var projectedOutflows = Round(
+            input.OpenBills
+                .Where(x => x.OutstandingAmount > 0m && x.DueUtc.Date <= horizonEndDate)
+                .Sum(x => x.OutstandingAmount) +
+            input.RecurringOutflows
+                .Where(x => x.Amount > 0m && x.DueUtc.Date <= horizonEndDate)
+                .Sum(x => x.Amount));
+
+        return new FinanceCashProjectionDto(
+            horizonDays,
+            Round(input.CurrentCash),
+            projectedInflows,
+            projectedOutflows,
+            Round(input.CurrentCash + projectedInflows - projectedOutflows));
+    }
+
+    private static FinanceObligationCoverageDto BuildObligationCoverage(
+        FinanceIntelligenceInputDto input,
+        FinanceCashProjectionDto sevenDayProjection)
+    {
+        var nearTermObligations = sevenDayProjection.ProjectedOutflows;
+        var availableCash = Round(input.CurrentCash + sevenDayProjection.ProjectedInflows);
+        var coverageRatio = nearTermObligations <= 0m
+            ? 999m
+            : Round(availableCash / nearTermObligations, 4);
+
+        var severity = nearTermObligations <= 0m
+            ? "healthy"
+            : sevenDayProjection.EndingCash < 0m || coverageRatio < 0.75m
+                ? "critical"
+                : coverageRatio < 1m
+                    ? "at_risk"
+                    : coverageRatio < 1.25m
+                        ? "watch"
+                        : "healthy";
+
+        var recommendation = severity switch
+        {
+            "critical" => new Recommendation("coverage_critical", "Projected cash turns negative inside the next 7 days. Delay non-critical bills immediately and escalate collections today."),
+            "at_risk" => new Recommendation("coverage_at_risk", "Near-term obligations are not fully covered over the next 7 days. Delay non-critical bills and accelerate collections today."),
+            "watch" => new Recommendation("coverage_watch_cash_buffer", "Near-term obligations are covered with a thin 7-day buffer. Delay non-urgent bills and confirm expected receipts."),
+            _ when nearTermObligations <= 0m => new Recommendation("coverage_clear", "No bills or recurring outflows fall due in the next 7 days."),
+            _ => new Recommendation("coverage_healthy", "Near-term obligations are covered for the next 7 days. Pay overdue bills now and keep collections moving.")
+        };
+
+        return new FinanceObligationCoverageDto(
+            7,
+            availableCash,
+            nearTermObligations,
+            coverageRatio,
+            severity,
+            recommendation.Code,
+            recommendation.Text);
+    }
+
+    private static IReadOnlyList<FinanceOverdueInvoiceRecommendationDto> BuildOverdueInvoices(FinanceIntelligenceInputDto input)
+    {
+        var asOfDate = input.AsOfUtc.Date;
+        var ranked = input.OpenInvoices
+            .Where(x => x.OutstandingAmount > 0m && x.DueUtc.Date < asOfDate)
+            .Select(x => new
+            {
+                Invoice = x,
+                OverdueDays = Math.Max(1, (int)(asOfDate - x.DueUtc.Date).TotalDays),
+                Severity = ResolveInvoiceSeverity(Math.Max(1, (int)(asOfDate - x.DueUtc.Date).TotalDays), x.OutstandingAmount)
+            })
+            .OrderByDescending(x => x.OverdueDays)
+            .ThenByDescending(x => x.Invoice.OutstandingAmount)
+            .ThenBy(x => x.Invoice.InvoiceNumber, StringComparer.OrdinalIgnoreCase)
+            .ThenBy(x => x.Invoice.InvoiceId)
+            .Select((x, index) =>
+            {
+                var recommendation = BuildInvoiceRecommendation(x.Invoice, x.OverdueDays, x.Severity);
+                return new FinanceOverdueInvoiceRecommendationDto(
+                    index + 1,
+                    x.Invoice.InvoiceId,
+                    x.Invoice.InvoiceNumber,
+                    x.Invoice.CustomerName,
+                    x.Invoice.DueUtc,
+                    Round(x.Invoice.OutstandingAmount),
+                    NormalizeCurrency(x.Invoice.Currency, input.Currency),
+                    x.OverdueDays,
+                    x.Severity,
+                    recommendation.Code,
+                    recommendation.Text);
+            })
+            .ToArray();
+
+        return ranked;
+    }
+
+    private static IReadOnlyList<FinanceDueSoonBillRecommendationDto> BuildDueSoonBills(
+        FinanceIntelligenceInputDto input,
+        FinanceCashProjectionDto sevenDayProjection)
+    {
+        var asOfDate = input.AsOfUtc.Date;
+        var horizonEndDate = asOfDate.AddDays(30);
+
+        return input.OpenBills
+            .Where(x => x.OutstandingAmount > 0m && x.DueUtc.Date <= horizonEndDate)
+            .Select(x =>
+            {
+                var daysUntilDue = (int)(x.DueUtc.Date - asOfDate).TotalDays;
+                var cashImpact = ResolveCashImpact(x.OutstandingAmount, input.CurrentCash);
+                var severity = ResolveBillSeverity(daysUntilDue, cashImpact);
+                return new
+                {
+                    Bill = x,
+                    DaysUntilDue = daysUntilDue,
+                    CashImpact = cashImpact,
+                    Severity = severity
+                };
+            })
+            .OrderBy(x => x.DaysUntilDue)
+            .ThenByDescending(x => CashImpactWeight(x.CashImpact))
+            .ThenByDescending(x => x.Bill.OutstandingAmount)
+            .ThenBy(x => x.Bill.BillNumber, StringComparer.OrdinalIgnoreCase)
+            .ThenBy(x => x.Bill.BillId)
+            .Select((x, index) =>
+            {
+                var recommendation = BuildBillRecommendation(x.Bill, x.DaysUntilDue, sevenDayProjection);
+                return new FinanceDueSoonBillRecommendationDto(
+                    index + 1,
+                    x.Bill.BillId,
+                    x.Bill.BillNumber,
+                    x.Bill.SupplierName,
+                    x.Bill.DueUtc,
+                    Round(x.Bill.OutstandingAmount),
+                    NormalizeCurrency(x.Bill.Currency, input.Currency),
+                    x.DaysUntilDue,
+                    x.CashImpact,
+                    x.Severity,
+                    recommendation.Code,
+                    recommendation.Text);
+            })
+            .ToArray();
+    }
+
+    private static string ResolveInvoiceSeverity(int overdueDays, decimal outstandingAmount) =>
+        overdueDays >= 45 || (overdueDays >= 30 && outstandingAmount >= 1000m)
+            ? "critical"
+            : overdueDays >= 14 || (overdueDays >= 7 && outstandingAmount >= 250m)
+                ? "at_risk"
+                : overdueDays >= 7
+                    ? "watch"
+                    : "healthy";
+
+    private static string ResolveBillSeverity(int daysUntilDue, string cashImpact) =>
+        daysUntilDue <= 0
+            ? "critical"
+            : daysUntilDue <= 3 || (string.Equals(cashImpact, "high", StringComparison.OrdinalIgnoreCase) && daysUntilDue <= 14)
+                ? "at_risk"
+                : daysUntilDue <= 14 || string.Equals(cashImpact, "high", StringComparison.OrdinalIgnoreCase)
+                    ? "watch"
+                    : "healthy";
+
+    private static string ResolveCashImpact(decimal outstandingAmount, decimal currentCash)
+    {
+        if (outstandingAmount <= 0m)
+        {
+            return "low";
+        }
+
+        if (currentCash <= 0m)
+        {
+            return "high";
+        }
+
+        var ratio = outstandingAmount / currentCash;
+        return ratio switch
+        {
+            >= 0.5m => "high",
+            >= 0.2m => "medium",
+            _ => "low"
+        };
+    }
+
+    private static Recommendation BuildInvoiceRecommendation(FinanceOpenReceivableItemDto invoice, int overdueDays, string severity)
+    {
+        var amount = FormatAmount(invoice.OutstandingAmount);
+        var currency = NormalizeCurrency(invoice.Currency, null);
+
+        return severity switch
+        {
+            "critical" => new Recommendation(
+                "call_customer_escalate",
+                $"Call {invoice.CustomerName} today about {invoice.InvoiceNumber}. It is {overdueDays} day(s) overdue for {amount} {currency}; escalate if no payment date is confirmed."),
+            "at_risk" => new Recommendation(
+                "call_customer_confirm_date",
+                $"Call {invoice.CustomerName} today about {invoice.InvoiceNumber} and confirm a payment date. It is {overdueDays} day(s) overdue for {amount} {currency}."),
+            _ => new Recommendation(
+                "send_payment_reminder",
+                $"Send a reminder today for {invoice.InvoiceNumber}. It is {overdueDays} day(s) overdue for {amount} {currency}.")
+        };
+    }
+
+    private static Recommendation BuildBillRecommendation(
+        FinanceOpenPayableItemDto bill,
+        int daysUntilDue,
+        FinanceCashProjectionDto sevenDayProjection)
+    {
+        var amount = FormatAmount(bill.OutstandingAmount);
+        var currency = NormalizeCurrency(bill.Currency, null);
+
+        if (daysUntilDue <= 0)
+        {
+            return new Recommendation(
+                "pay_now",
+                $"Pay now. {bill.BillNumber} is overdue by {Math.Abs(daysUntilDue)} day(s) and the outstanding amount is {amount} {currency}.");
+        }
+
+        if (daysUntilDue <= 3 || (daysUntilDue <= 7 && sevenDayProjection.EndingCash >= 0m))
+        {
+            return new Recommendation(
+                "pay_now",
+                $"Pay now. {bill.BillNumber} is due in {daysUntilDue} day(s) and the outstanding amount is {amount} {currency}.");
+        }
+
+        var reviewDate = bill.DueUtc.Date.AddDays(-2);
+        return new Recommendation(
+            "delay_and_recheck",
+            $"Delay. {bill.BillNumber} is due in {daysUntilDue} day(s); preserve cash for earlier obligations and review it again on {reviewDate:yyyy-MM-dd}.");
+    }
+
+    private static int CashImpactWeight(string value) =>
+        value switch
+        {
+            "high" => 2,
+            "medium" => 1,
+            _ => 0
+        };
+
+    private static string NormalizeCurrency(string? primaryCurrency, string? fallbackCurrency)
+    {
+        if (!string.IsNullOrWhiteSpace(primaryCurrency))
+        {
+            return primaryCurrency.Trim().ToUpperInvariant();
+        }
+
+        if (!string.IsNullOrWhiteSpace(fallbackCurrency))
+        {
+            return fallbackCurrency.Trim().ToUpperInvariant();
+        }
+
+        return "USD";
+    }
+
+    private static string FormatAmount(decimal amount) =>
+        Round(amount).ToString("0.00", CultureInfo.InvariantCulture);
+
+    private static decimal Round(decimal value, int decimals = 2) =>
+        Math.Round(value, decimals, MidpointRounding.AwayFromZero);
+
+    private sealed record Recommendation(string Code, string Text);
+}
