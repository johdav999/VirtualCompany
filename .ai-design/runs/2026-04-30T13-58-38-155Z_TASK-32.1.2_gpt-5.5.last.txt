+
+public sealed class DataProtectionFortnoxOAuthStateProtector : IFortnoxOAuthStateProtector
+{
+    private static readonly JsonSerializerOptions SerializerOptions = new(JsonSerializerDefaults.Web);
+    private readonly IDataProtector _protector;
+
+    public DataProtectionFortnoxOAuthStateProtector(IDataProtectionProvider dataProtectionProvider)
+    {
+        _protector = dataProtectionProvider.CreateProtector("VirtualCompany.FortnoxOAuthState.v1");
+    }
+
+    public string Protect(FortnoxOAuthState state)
+    {
+        ValidateState(state);
+        return _protector.Protect(JsonSerializer.Serialize(state, SerializerOptions));
+    }
+
+    public FortnoxOAuthState Unprotect(string protectedState)
+    {
+        if (string.IsNullOrWhiteSpace(protectedState))
+        {
+            throw new UnauthorizedAccessException("Fortnox OAuth state was invalid.");
+        }
+
+        try
+        {
+            var json = _protector.Unprotect(protectedState);
+            var state = JsonSerializer.Deserialize<FortnoxOAuthState>(json, SerializerOptions)
+                ?? throw new InvalidOperationException("Fortnox OAuth state was invalid.");
+
+            ValidateState(state);
+            return state;
+        }
+        catch (Exception ex) when (ex is CryptographicException or JsonException or NotSupportedException or ArgumentException or InvalidOperationException)
+        {
+            throw new UnauthorizedAccessException("Fortnox OAuth state was invalid.", ex);
+        }
+    }
+
+    private static void ValidateState(FortnoxOAuthState state)
+    {
+        if (state.CompanyId == Guid.Empty ||
+            state.UserId == Guid.Empty ||
+            string.IsNullOrWhiteSpace(state.Nonce) ||
+            state.IssuedUtc == default ||
+            state.ExpiresUtc == default ||
+            state.ExpiresUtc <= state.IssuedUtc)
+        {
+            throw new InvalidOperationException("Fortnox OAuth state was invalid.");
+        }
+    }

+    public async Task<FortnoxAccessTokenResult> GetValidAccessTokenAsync(
+        RefreshFortnoxAccessTokenCommand command,
+        CancellationToken cancellationToken)
+    {
+        var query = _dbContext.FortnoxConnections
+            .Where(x => x.CompanyId == command.CompanyId);
+
+        if (command.ConnectionId.HasValue)
+        {
+            query = query.Where(x => x.Id == command.ConnectionId.Value);
+        }
+
+        var connection = await query.SingleOrDefaultAsync(cancellationToken);
+        if (connection is null)
+        {
+            return FortnoxAccessTokenResult.ReconnectRequired("Fortnox is not connected.");
+        }
+
+        if (connection.Status is FortnoxConnectionStatus.NeedsReconnect or FortnoxConnectionStatus.Revoked or FortnoxConnectionStatus.Disconnected)
+        {
+            return FortnoxAccessTokenResult.ReconnectRequired("Fortnox needs to be reconnected.");
+        }
+
+        var now = UtcNow();
+        if (!string.IsNullOrWhiteSpace(connection.EncryptedAccessToken) &&
+            (!connection.AccessTokenExpiresUtc.HasValue || connection.AccessTokenExpiresUtc.Value > now.Add(RefreshSkew)))
+        {
+            return FortnoxAccessTokenResult.Success(
+                _fieldEncryption.Decrypt(command.CompanyId, AccessTokenPurpose, connection.EncryptedAccessToken),
+                connection.AccessTokenExpiresUtc);
+        }
+
+        if (string.IsNullOrWhiteSpace(connection.EncryptedRefreshToken))
+        {
+            connection.SetStatus(FortnoxConnectionStatus.NeedsReconnect, "Fortnox needs to be reconnected.", now);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+            return FortnoxAccessTokenResult.ReconnectRequired("Fortnox needs to be reconnected.");
+        }
+
+        connection.RecordRefreshAttempt(now);
+        var refreshToken = _fieldEncryption.Decrypt(command.CompanyId, RefreshTokenPurpose, connection.EncryptedRefreshToken);
+        try
+        {
+            var tokenResult = await _client.RefreshTokenAsync(refreshToken, cancellationToken);
+            connection.StoreRefreshedTokens(
+                _fieldEncryption.Encrypt(command.CompanyId, AccessTokenPurpose, tokenResult.AccessToken),
+                _fieldEncryption.Encrypt(command.CompanyId, RefreshTokenPurpose, tokenResult.RefreshToken),
+                tokenResult.AccessTokenExpiresUtc,
+                tokenResult.GrantedScopes,
+                now);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+
+            return FortnoxAccessTokenResult.Success(tokenResult.AccessToken, tokenResult.AccessTokenExpiresUtc);
+        }
+        catch (FortnoxOAuthException ex) when (ex.RequiresReconnect)
+        {
+            connection.SetStatus(FortnoxConnectionStatus.NeedsReconnect, ex.SafeMessage, now);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+            _logger.LogWarning(
+                "Fortnox refresh token rejected. CompanyId: {CompanyId}. ConnectionId: {ConnectionId}.",
+                command.CompanyId,
+                connection.Id);
+            return FortnoxAccessTokenResult.ReconnectRequired(ex.SafeMessage);
+        }
+        catch (Exception ex) when (ex is FortnoxOAuthException or HttpRequestException or TaskCanceledException)
+        {
+            connection.SetStatus(FortnoxConnectionStatus.Error, "Fortnox token refresh failed. The job will retry later.", now);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+            _logger.LogWarning(
+                "Fortnox token refresh failed without exposing token material. CompanyId: {CompanyId}. ConnectionId: {ConnectionId}.",
+                command.CompanyId,
+                connection.Id);
+            return FortnoxAccessTokenResult.TransientFailure("Fortnox token refresh failed. The job will retry later.");
+        }
+    }
+
+    public async Task<FortnoxConnectionStatusResult> GetStatusAsync(
+        GetFortnoxConnectionStatusQuery query,
+        CancellationToken cancellationToken)
+    {
+        EnsureCurrentTenantUser(query.CompanyId, query.UserId);
+        var connection = await _dbContext.FortnoxConnections
+            .AsNoTracking()
+            .SingleOrDefaultAsync(x => x.CompanyId == query.CompanyId, cancellationToken);
+
+        if (connection is null)
+        {
+            return new FortnoxConnectionStatusResult(false, null, null, null, null, null, null);
+        }
+
+        return new FortnoxConnectionStatusResult(
+            connection.Status == FortnoxConnectionStatus.Connected,
+            connection.Id,
+            connection.Status.ToStorageValue(),
+            connection.ConnectedUtc,
+            connection.AccessTokenExpiresUtc,
+            connection.LastRefreshAttemptUtc,
+            connection.LastErrorSummary);
+    }
+
+    public async Task MarkNeedsReconnectAsync(
+        Guid companyId,
+        Guid connectionId,
+        string safeReason,
+        CancellationToken cancellationToken)
+    {
+        var connection = await _dbContext.FortnoxConnections
+            .SingleOrDefaultAsync(x => x.CompanyId == companyId && x.Id == connectionId, cancellationToken);
+
+        if (connection is null)
+        {
+            return;
+        }
+
+        connection.SetStatus(FortnoxConnectionStatus.NeedsReconnect, safeReason, UtcNow());
+        await _dbContext.SaveChangesAsync(cancellationToken);
+    }
+
+    private void ValidateCallbackState(CompleteFortnoxOAuthConnectionCommand command, FortnoxOAuthState state)
+    {
+        var now = UtcNow();
+        if (state.ExpiresUtc <= now)
+        {
+            throw new FortnoxOAuthException("Fortnox authorization has expired. Start the connection again.");
+        }
+
+        if (state.CompanyId != command.CompanyId || state.UserId != command.UserId)
+        {
+            throw new UnauthorizedAccessException("Fortnox authorization did not match the current company and user.");
+        }
+
+        if (!string.IsNullOrWhiteSpace(command.Nonce) &&
+            !CryptographicOperations.FixedTimeEquals(
+                System.Text.Encoding.UTF8.GetBytes(state.Nonce),
+                System.Text.Encoding.UTF8.GetBytes(command.Nonce)))
+        {
+            throw new UnauthorizedAccessException("Fortnox authorization nonce was invalid.");
+        }
+    }
+
+    private void EnsureCurrentTenantUser(Guid companyId, Guid userId)
+    {
+        if (!_companyContextAccessor.IsResolved ||
+            _companyContextAccessor.CompanyId != companyId ||
+            _companyContextAccessor.UserId != userId)
+        {
+            throw new UnauthorizedAccessException("Fortnox connections are scoped to the current tenant and user.");
+        }
+    }
+
+    private DateTime UtcNow() => _timeProvider.GetUtcNow().UtcDateTime;
+
+    private static string CreateNonce()
+    {
+        Span<byte> bytes = stackalloc byte[32];
+        RandomNumberGenerator.Fill(bytes);
+        return Convert.ToBase64String(bytes);
+    }

+# Fortnox integration
+
+Virtual Company connects to Fortnox with OAuth 2.0 authorization code flow. The integration is tenant-scoped: each company has one Fortnox connection record and encrypted token material.
+
+## Fortnox app registration
+
+Create an app in Fortnox Developer and configure the redirect URI to match the deployed application:
+
+```text
+https://{host}/finance/integrations/fortnox/callback
+```
+
+For local development use the same path on the local web host, for example:
+
+```text
+https://localhost:7136/finance/integrations/fortnox/callback
+```

+
+## Connect flow
+
+1. A company admin starts the flow from `/finance/integrations/fortnox/connect`.
+2. The server creates a protected state payload containing company id, user id, nonce, issue time, expiry, reconnect intent, and optional return URL.
+3. The user is redirected to Fortnox with `client_id`, `redirect_uri`, `response_type=code`, configured scopes, `state`, and `nonce`.
+4. Fortnox redirects back to `/finance/integrations/fortnox/callback`.
+5. The server validates the authenticated user, company scope, state, expiry, and nonce when supplied.
+6. The authorization code is exchanged at `/oauth-v1/token`.
+7. Access and refresh tokens are encrypted before persistence.
+
+Token values are never returned by the API and must not be logged.

+# Fortnox integration runbook
+
+## Secret storage
+
+Store `FinanceIntegrations:Fortnox:ClientSecret` in the production secret store. Do not place Fortnox client secrets in checked-in configuration, container images, or deployment logs.
+
+The application encrypts Fortnox access and refresh tokens with ASP.NET Core Data Protection through the field encryption service. Production must persist Data Protection keys in the approved shared key store for the environment.
+
+## Startup validation
+
+When `FinanceIntegrations:Fortnox:Enabled` is `true`, the application fails startup if a required value is missing or not an absolute HTTP or HTTPS URI. Disable Fortnox to run the app without Fortnox secrets.
+
+Required keys:
+
+- `ClientId`
+- `ClientSecret`
+- `RedirectUri`
+- `AuthorizationUrl`
+- `TokenUrl`
+- `ApiBaseUrl`
+
+## Operational states
+
+Fortnox connection statuses:
+
+- `connected`: token material is present and usable.
+- `needs_reconnect`: refresh token was rejected or token material is missing.
+- `revoked`: provider revocation was detected.
+- `error`: transient token refresh failure; background jobs should retry later.
+- `disconnected`: operator disconnected the integration.
+
+Show plain English messages in UI. Do not expose enum names directly to users.
+
+## Token refresh
+
+Call `IFortnoxOAuthService.GetValidAccessTokenAsync` before Fortnox API calls. The service refreshes access tokens when they are expired or within the refresh window. It returns a structured result so background jobs can record a safe failure and continue without crashing the worker.
+
+If Fortnox rejects a refresh token with authorization failure, the service marks the connection as `needs_reconnect` and returns a reconnect-required result.
+
+## Troubleshooting
+
+Missing configuration:
+
+- Check the `FinanceIntegrations:Fortnox` section.
+- Confirm Fortnox is disabled in environments that should not connect.
+
+Callback fails:
+
+- Confirm the Fortnox app redirect URI exactly matches `/finance/integrations/fortnox/callback`.
+- Confirm the user is still signed in and belongs to the same company that initiated the flow.
+- Retry if the state expired; state is intentionally short-lived.
+
+Refresh fails:
+
+- `needs_reconnect` means an admin must run the Fortnox connect flow again.
+- `error` means retry later and check Fortnox availability.
+
+## Logging expectations
+
+Logs may include company id, user id, connection id, provider status category, and safe error category. Logs must never include access tokens, refresh tokens, authorization codes, client secrets, or raw provider response bodies.
