+                    <div class="template-grid mt-4">
+                        @foreach (var membership in ActiveMemberships)
+                        {

+else
+{
+    <section class="py-4 dashboard-hero">
+        <div class="dashboard-shell">
+            <div class="wizard-card dashboard-summary">
+                <span class="dashboard-kicker">Approval inbox</span>
+                <h1>@(companyAccess?.CompanyName ?? "Approvals")</h1>

+                @if (!string.IsNullOrWhiteSpace(errorMessage))

+@code {
+    [SupplyParameterFromQuery(Name = "companyId")]
+    public Guid? CompanyId { get; set; }
+
+    [SupplyParameterFromQuery(Name = "approvalId")]
+    public Guid? ApprovalId { get; set; }
+
+    private bool isLoading = true;
+    private bool isDeciding;
+    private CurrentUserContextViewModel? currentUser;
+    private CompanyAccessViewModel? companyAccess;
+    private List<ApprovalRequestViewModel> approvals = [];
+    private ApprovalRequestViewModel? selectedApproval;
+    private string? decisionComment;
+    private string? errorMessage;
+    private IReadOnlyList<CompanyMembershipViewModel> ActiveMemberships =>
+        currentUser?.Memberships
+            .Where(x => string.Equals(x.Status, "active", StringComparison.OrdinalIgnoreCase))
+            .ToList()
+        ?? [];
+
+    protected override async Task OnParametersSetAsync()
+    {
+        isLoading = true;
+        errorMessage = null;
+
+        try
+        {
+            currentUser = await OnboardingApiClient.GetCurrentUserContextAsync();
+            if (CompanyId is not Guid companyId)
+            {
+                if (currentUser?.ActiveCompany is not null)
+                {
+                    Navigation.NavigateTo($"/approvals?companyId={currentUser.ActiveCompany.CompanyId}", replace: true);
+                }
+
+                return;
+            }
+
+            companyAccess = await OnboardingApiClient.GetCompanyAccessAsync(companyId);
+            approvals = (await ApprovalApiClient.ListAsync(companyId)).ToList();
+            selectedApproval = ApprovalId is Guid approvalId
+                ? await ApprovalApiClient.GetAsync(companyId, approvalId)
+                : approvals.FirstOrDefault();
+        }
+        catch (OnboardingApiException ex)
+        {
+            errorMessage = ex.Message;
+            approvals = [];
+            selectedApproval = null;
+        }
+        finally
+        {
+            isLoading = false;
+        }
+    }
+
+    private async Task SelectApprovalAsync(Guid approvalId)
+    {
+        if (CompanyId is not Guid companyId)
+        {
+            return;
+        }
+
+        selectedApproval = await ApprovalApiClient.GetAsync(companyId, approvalId);
+        decisionComment = null;
+        Navigation.NavigateTo($"/approvals?companyId={companyId}&approvalId={approvalId}", replace: true);
+    }
+
+    private async Task DecideAsync(string decision)
+    {
+        if (CompanyId is not Guid companyId || selectedApproval?.CurrentStep is null)
+        {
+            return;
+        }
+
+        isDeciding = true;
+        errorMessage = null;
+        try
+        {
+            var result = await ApprovalApiClient.DecideAsync(
+                companyId,
+                selectedApproval.Id,
+                new ApprovalDecisionRequest
+                {
+                    ApprovalId = selectedApproval.Id,
+                    StepId = selectedApproval.CurrentStep.Id,
+                    Decision = decision,
+                    Comment = decisionComment
+                });
+            selectedApproval = result.Approval;
+            approvals = (await ApprovalApiClient.ListAsync(companyId)).ToList();
+        }
+        catch (OnboardingApiException ex)
+        {
+            errorMessage = ex.Message;
+        }
+        finally
+        {
+            isDeciding = false;
+        }
+    }
+
+    private string GetApprovalListItemClass(ApprovalRequestViewModel approval) =>
+        selectedApproval?.Id == approval.Id
+            ? "list-group-item list-group-item-action active"
+            : "list-group-item list-group-item-action";
+
+    private static bool IsPending(ApprovalRequestViewModel approval) =>
+        string.Equals(approval.Status, "pending", StringComparison.OrdinalIgnoreCase);
