+    public static string ToStorageValue(this FinanceSeedingState state) =>
+        state switch
+        {
+            FinanceSeedingState.NotSeeded => "not_seeded",
+            FinanceSeedingState.Seeding => "seeding",
+            FinanceSeedingState.Seeded => "seeded",
+            FinanceSeedingState.Failed => "failed",
+            _ => throw new ArgumentOutOfRangeException(nameof(state), state, "Unsupported finance seeding state.")

+    public static FinanceSeedingState Parse(string value)
+    {
+        if (TryParse(value, out var state))
+        {
+            return state;
+        }
+
+        throw new ArgumentOutOfRangeException(nameof(value), value, "Unsupported finance seeding state value.");
+    }
+
+    public static bool TryParse(string? value, out FinanceSeedingState state)
+    {
+        state = FinanceSeedingState.NotSeeded;
+        if (string.IsNullOrWhiteSpace(value))
+        {
+            return false;
+        }
+
+        return value.Trim().ToLowerInvariant() switch
+        {
+            "not_seeded" => (state = FinanceSeedingState.NotSeeded) is FinanceSeedingState,
+            "seeding" or "partially_seeded" => (state = FinanceSeedingState.Seeding) is FinanceSeedingState,
+            "seeded" or "fully_seeded" => (state = FinanceSeedingState.Seeded) is FinanceSeedingState,
+            "failed" => (state = FinanceSeedingState.Failed) is FinanceSeedingState,
+            _ => Enum.TryParse(value.Trim(), ignoreCase: true, out state)
+        };
+    }
+
+    public static void EnsureSupported(FinanceSeedingState state, string? paramName = null)
+    {
+        if (!Enum.IsDefined(state))
+        {
+            throw new ArgumentOutOfRangeException(paramName ?? nameof(state), state, "Unsupported finance seeding state.");
+        }
+    }
+

+        if (string.IsNullOrWhiteSpace(columnName))

+        }
+

+        FinanceSeedingState persistedState,
+        FinanceSeedingMetadataSnapshot? metadata,
+        FinanceSeedRecordChecks recordChecks,

+    {
+        if (recordChecks.IsComplete)

+            return "Metadata indicates finance seeding started, but foundational records could not be fully confirmed yet.";
+        }

+    private async Task<(FinanceSeedingStateResultDto State, BackgroundExecution? Execution, bool SeedJobEnqueued)> EnsureSeedRequestedCoreAsync(
+        GetFinanceEntryStateQuery query,
+        DateTime checkedAtUtc,
+        CancellationToken cancellationToken)
+    {
+        var company = await _dbContext.Companies
+            .IgnoreQueryFilters()
+            .SingleAsync(x => x.Id == query.CompanyId, cancellationToken);
+
+        var execution = await FindExecutionAsync(query.CompanyId, cancellationToken);
+        var seedingState = await _financeSeedingStateService.GetCompanyFinanceSeedingStateAsync(query.CompanyId, cancellationToken);
+        var seedJobEnqueued = false;
+
+        if (ShouldRetry(query, seedingState.State, execution))
+        {
+            FinanceSeedingMetadata.MarkSeeding(company, checkedAtUtc, requestedAtUtc: checkedAtUtc);
+            if (execution is null)
+            {
+                execution = CreateExecution(query.CompanyId);
+                _dbContext.BackgroundExecutions.Add(execution);
+            }
+            else
+            {
+                execution.Queue(
+                    checkedAtUtc,
+                    _identityFactory.EnsureCorrelationId(_correlationContextAccessor?.CorrelationId),
+                    resetAttempts: true);
+            }
+
+            seedJobEnqueued = true;
+            await WriteRequestedAuditAsync(query.CompanyId, execution, query.Source, isRetry: true, cancellationToken);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+        }
+        else if (ShouldEnqueueInitialRequest(seedingState.State, execution))
+        {
+            FinanceSeedingMetadata.MarkSeeding(company, checkedAtUtc, requestedAtUtc: checkedAtUtc);
+            execution = CreateExecution(query.CompanyId);
+            _dbContext.BackgroundExecutions.Add(execution);
+            seedJobEnqueued = true;
+            await WriteRequestedAuditAsync(query.CompanyId, execution, query.Source, isRetry: false, cancellationToken);
+            await _dbContext.SaveChangesAsync(cancellationToken);
+            _logger.LogInformation(
+                "Queued finance seed background execution {ExecutionId} for company {CompanyId}.",
+                execution.Id,
+                query.CompanyId);
+        }
+
+        var refreshedState = await _financeSeedingStateService.GetCompanyFinanceSeedingStateAsync(query.CompanyId, cancellationToken);
+        var refreshedExecution = await FindExecutionAsync(query.CompanyId, cancellationToken);
+        return (refreshedState, refreshedExecution, seedJobEnqueued);
+    }

+            _ =>
+                "Finance data is being prepared in the background. This page refreshes automatically when setup completes."
+        };
