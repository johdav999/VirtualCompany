+    public async Task Canonical_decision_endpoint_approve_updates_approval_task_step_and_audit_state()
+    {
+        var seed = await SeedPendingTaskApprovalAsync("approve");
+
+        using var client = CreateAuthenticatedClient(seed.ApproverSubject, seed.ApproverEmail, "Finance Approver");
+        var response = await client.PostAsJsonAsync($"/api/companies/{seed.CompanyId}/approvals/{seed.ApprovalId}/decisions", new
+        {
+            decision = "approve",
+            stepId = seed.StepId,
+            comment = "Approved from mobile."
+        });
+
+        Assert.Equal(HttpStatusCode.OK, response.StatusCode);
+        var result = await response.Content.ReadFromJsonAsync<ApprovalDecisionResultDto>();
+        Assert.NotNull(result);
+        Assert.True(result!.IsFinalized);
+        Assert.Equal("approved", result.Approval.Status);
+        Assert.Equal("approved", result.DecidedStep.Status);
+        Assert.Equal("Approved from mobile.", result.DecidedStep.Comment);
+        Assert.Null(result.NextStep);
+
+        using var scope = _factory.Services.CreateScope();
+        var dbContext = scope.ServiceProvider.GetRequiredService<VirtualCompanyDbContext>();
+        var approval = await dbContext.ApprovalRequests
+            .AsNoTracking()
+            .Include(x => x.Steps)
+            .SingleAsync(x => x.Id == seed.ApprovalId);
+        var task = await dbContext.WorkTasks.AsNoTracking().SingleAsync(x => x.Id == seed.TaskId);
+        var auditActions = await dbContext.AuditEvents
+            .AsNoTracking()
+            .Where(x => x.CompanyId == seed.CompanyId && x.RelatedApprovalRequestId == seed.ApprovalId)
+            .Select(x => x.Action)
+            .ToListAsync();
+
+        Assert.Equal(ApprovalRequestStatus.Approved, approval.Status);
+        Assert.Equal("Approval chain completed.", approval.DecisionSummary);
+        Assert.NotNull(approval.DecidedUtc);
+        var step = Assert.Single(approval.Steps);
+        Assert.Equal(ApprovalStepStatus.Approved, step.Status);
+        Assert.Equal(seed.ApproverUserId, step.DecidedByUserId);
+        Assert.Equal("Approved from mobile.", step.Comment);
+        Assert.NotNull(step.DecidedUtc);
+        Assert.Equal(WorkTaskStatus.InProgress, task.Status);
+        Assert.Contains(AuditEventActions.ApprovalStepApproved, auditActions);
+        Assert.Contains(AuditEventActions.ApprovalCompleted, auditActions);
+        Assert.Contains(AuditEventActions.ApprovalLinkedEntityStateUpdated, auditActions);
+    }

+    private async Task<ApprovalDecisionSeed> SeedPendingTaskApprovalAsync(string suffix)
+    {
+        var companyId = Guid.NewGuid();
+        var otherCompanyId = Guid.NewGuid();
+        var approverUserId = Guid.NewGuid();
+        var observerUserId = Guid.NewGuid();
+        var requestedByAgentId = Guid.NewGuid();
+        var taskId = Guid.NewGuid();
+        var approvalId = Guid.NewGuid();
+        var approverSubject = $"approval-{suffix}-approver-{Guid.NewGuid():N}";
+        var observerSubject = $"approval-{suffix}-observer-{Guid.NewGuid():N}";
+        var approverEmail = $"{approverSubject}@example.com";
+        var observerEmail = $"{observerSubject}@example.com";
+        Guid stepId = Guid.Empty;
+
+        await _factory.SeedAsync(dbContext =>
+        {
+            dbContext.Users.AddRange(
+                new User(approverUserId, approverEmail, "Finance Approver", "dev-header", approverSubject),
+                new User(observerUserId, observerEmail, "Observer", "dev-header", observerSubject));
+            dbContext.Companies.AddRange(
+                new Company(companyId, "Approval Company"),
+                new Company(otherCompanyId, "Other Approval Company"));
+            dbContext.CompanyMemberships.AddRange(
+                new CompanyMembership(Guid.NewGuid(), companyId, approverUserId, CompanyMembershipRole.FinanceApprover, CompanyMembershipStatus.Active),
+                new CompanyMembership(Guid.NewGuid(), companyId, observerUserId, CompanyMembershipRole.Employee, CompanyMembershipStatus.Active),
+                new CompanyMembership(Guid.NewGuid(), otherCompanyId, approverUserId, CompanyMembershipRole.Owner, CompanyMembershipStatus.Active));
+
+            var task = new WorkTask(
+                taskId,
+                companyId,
+                "orchestration",
+                $"Approval task {suffix}",
+                "Requires approval.",
+                WorkTaskPriority.High,
+                null,
+                null,
+                "agent",
+                requestedByAgentId,
+                new Dictionary<string, JsonNode?> { ["amount"] = JsonValue.Create(25000) });
+            task.UpdateStatus(WorkTaskStatus.AwaitingApproval);
+
+            var approval = ApprovalRequest.CreateForTarget(
+                approvalId,
+                companyId,
+                ApprovalTargetEntityType.Task,
+                taskId,
+                "agent",
+                requestedByAgentId,
+                "threshold",
+                new Dictionary<string, JsonNode?>
+                {
+                    ["thresholdKey"] = JsonValue.Create("amount"),
+                    ["thresholdValue"] = JsonValue.Create("25000"),
+                    ["configuredThreshold"] = JsonValue.Create("10000")
+                },
+                CompanyMembershipRole.FinanceApprover.ToStorageValue(),
+                null,
+                []);
+            stepId = approval.CurrentActionableStep!.Id;
+
+            dbContext.WorkTasks.Add(task);
+            dbContext.ApprovalRequests.Add(approval);
+            return Task.CompletedTask;
+        });
+
+        return new ApprovalDecisionSeed(companyId, otherCompanyId, approvalId, stepId, taskId, approverUserId, approverSubject, approverEmail, observerSubject, observerEmail);
+    }
