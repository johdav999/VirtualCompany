+    {

+
+        using var companyAClient = CreateAuthenticatedClient(factory, seed.Subject, seed.Email, seed.DisplayName, seed.CompanyAId);

+
+        var request = new StartCompanySimulationRequest(
+            new DateTime(2026, 4, 1, 0, 0, 0, DateTimeKind.Utc),
+            GenerationEnabled: true,
+            Seed: 73,
+            DeterministicConfigurationJson: """{"financeGeneration":{"anomalyCadenceDays":3,"anomalyOffsetDays":1}}""");
+
+        var startCompanyA = await companyAClient.PostAsJsonAsync($"/internal/companies/{seed.CompanyAId}/simulation/start", request);
+        var startCompanyB = await companyBClient.PostAsJsonAsync($"/internal/companies/{seed.CompanyBId}/simulation/start", request);

+        Assert.Equal(HttpStatusCode.Created, startCompanyB.StatusCode);
+
+        timeProvider.Advance(TimeSpan.FromSeconds(100));
+        _ = await companyAClient.GetFromJsonAsync<CompanySimulationStateDto>($"/internal/companies/{seed.CompanyAId}/simulation");
+        _ = await companyBClient.GetFromJsonAsync<CompanySimulationStateDto>($"/internal/companies/{seed.CompanyBId}/simulation");
+
+        var firstSnapshot = await CaptureSnapshotAsync(factory, companyAClient, seed.CompanyAId);
+        var secondSnapshot = await CaptureSnapshotAsync(factory, companyBClient, seed.CompanyBId);
+
+        Assert.Equal(firstSnapshot.InvoiceRows, secondSnapshot.InvoiceRows);
+        Assert.Equal(firstSnapshot.BillRows, secondSnapshot.BillRows);
+        Assert.Equal(firstSnapshot.TransactionRows, secondSnapshot.TransactionRows);
+        Assert.Equal(firstSnapshot.ApprovalRows, secondSnapshot.ApprovalRows);
+        Assert.Equal(firstSnapshot.AlertTypes, secondSnapshot.AlertTypes);
+        Assert.Equal(firstSnapshot.AlertDates, secondSnapshot.AlertDates);
+        Assert.Equal(firstSnapshot.AuditActions, secondSnapshot.AuditActions);
+    }

+    private static async Task<FinanceReplaySnapshot> CaptureSnapshotAsync(
+        TestWebApplicationFactory factory,
+        HttpClient client,
+        Guid companyId)
+    {
+        var invoices = await GetArrayAsync(client, $"/internal/companies/{companyId}/finance/invoices?limit=200");
+        var bills = await GetArrayAsync(client, $"/internal/companies/{companyId}/finance/bills?limit=200");
+        var transactions = await GetArrayAsync(client, $"/internal/companies/{companyId}/finance/transactions?limit=200");
+
+        var persistence = await factory.ExecuteDbContextAsync(async dbContext =>
+        {
+            var approvalRows = await dbContext.ApprovalRequests
+                .IgnoreQueryFilters()
+                .Where(x => x.CompanyId == companyId)
+                .OrderBy(x => x.ApprovalType)
+                .ThenBy(x => x.Status)
+                .Select(x => $"{x.ApprovalType}|{x.Status}")
+                .ToListAsync();
+            var alertTypes = await dbContext.Alerts
+                .IgnoreQueryFilters()
+                .Where(x => x.CompanyId == companyId && x.Type == AlertType.Anomaly)
+                .OrderBy(x => x.Fingerprint)
+                .Select(x => x.Metadata["anomalyType"]!.GetValue<string>())
+                .ToListAsync();
+            var alertDates = await dbContext.Alerts
+                .IgnoreQueryFilters()
+                .Where(x => x.CompanyId == companyId && x.Type == AlertType.Anomaly)
+                .OrderBy(x => x.CreatedUtc)
+                .Select(x => (x.LastDetectedUtc ?? x.CreatedUtc).Date.ToString("yyyy-MM-dd", CultureInfo.InvariantCulture))
+                .Distinct()
+                .ToListAsync();
+            var auditActions = await dbContext.AuditEvents
+                .IgnoreQueryFilters()
+                .Where(x => x.CompanyId == companyId && x.Action.StartsWith("finance.", StringComparison.OrdinalIgnoreCase))
+                .OrderBy(x => x.Action)
+                .Select(x => x.Action)
+                .ToListAsync();
+
+            return new
+            {
+                ApprovalRows = approvalRows,
+                AlertTypes = alertTypes,
+                AlertDates = alertDates,
+                AuditActions = auditActions
+            };
+        });
+
+        return new FinanceReplaySnapshot(
+            invoices
+                .Select(node => node!.AsObject())
+                .Select(row => $"{GetRequiredString(row, "invoiceNumber")}|{GetRequiredDecimal(row, "amount").ToString("0.00", CultureInfo.InvariantCulture)}|{GetRequiredString(row, "currency")}|{GetRequiredString(row, "status")}")
+                .OrderBy(x => x, StringComparer.Ordinal)
+                .ToList(),
+            bills
+                .Select(node => node!.AsObject())
+                .Select(row => $"{GetRequiredString(row, "billNumber")}|{GetRequiredDecimal(row, "amount").ToString("0.00", CultureInfo.InvariantCulture)}|{GetRequiredString(row, "currency")}|{GetRequiredString(row, "status")}")
+                .OrderBy(x => x, StringComparer.Ordinal)
+                .ToList(),
+            transactions
+                .Select(node => node!.AsObject())
+                .Select(row => $"{GetRequiredString(row, "externalReference")}|{GetRequiredString(row, "transactionType")}|{GetRequiredDecimal(row, "amount").ToString("0.00", CultureInfo.InvariantCulture)}|{GetRequiredString(row, "currency")}|{GetRequiredBool(row, "isFlagged")}")
+                .OrderBy(x => x, StringComparer.Ordinal)
+                .ToList(),
+            persistence.ApprovalRows,
+            persistence.AlertTypes,
+            persistence.AlertDates,
+            persistence.AuditActions);
+    }

+    private static async Task<SimulationFinanceSeed> SeedSimulationFinanceCompanyAsync(TestWebApplicationFactory factory)
+    {
+        var ownerUserId = Guid.NewGuid();
+        var approverUserId = Guid.NewGuid();
+        var companyId = Guid.NewGuid();
+        var otherCompanyId = Guid.NewGuid();
+        var ownerSubject = $"sim-finance-owner-{Guid.NewGuid():N}";
+        var approverSubject = $"sim-finance-approver-{Guid.NewGuid():N}";
+        var ownerEmail = $"{ownerSubject}@example.com";

+
+        await factory.SeedAsync(dbContext =>

+                new User(ownerUserId, ownerEmail, "Simulation Finance Owner", "dev-header", ownerSubject),

+                new Company(companyId, "Simulation Finance Company"),
+                new Company(otherCompanyId, "Other Simulation Finance Company"));

+                CreatePolicy(companyId),
+                CreatePolicy(otherCompanyId));
+            return Task.CompletedTask;
+        });
+
+        return new SimulationFinanceSeed(
+            companyId,
+            otherCompanyId,
+            ownerSubject,
+            ownerEmail,
+            "Simulation Finance Owner",
+            approverSubject,
+            approverEmail,
