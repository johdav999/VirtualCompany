+    private async Task<FinanceSandboxDatasetGenerationResponse?> BuildSandboxDatasetGenerationAsync(
+        Guid companyId,
+        CancellationToken cancellationToken)
+    {
+        var transactionCount = await _dbContext.FinanceTransactions
+            .Where(x => x.CompanyId == companyId)
+            .CountAsync(cancellationToken);
+        var invoiceCount = await _dbContext.FinanceInvoices
+            .Where(x => x.CompanyId == companyId)
+            .CountAsync(cancellationToken);
+        var balanceCount = await _dbContext.FinanceBalances
+            .Where(x => x.CompanyId == companyId)
+            .CountAsync(cancellationToken);
+
+        if (transactionCount == 0 && invoiceCount == 0 && balanceCount == 0)
+        {
+            return null;
+        }
+
+        var lastTransactionUtc = await _dbContext.FinanceTransactions
+            .Where(x => x.CompanyId == companyId)
+            .OrderByDescending(x => x.CreatedUtc)
+            .Select(x => (DateTime?)x.CreatedUtc)
+            .FirstOrDefaultAsync(cancellationToken);
+        var lastInvoiceUtc = await _dbContext.FinanceInvoices
+            .Where(x => x.CompanyId == companyId)
+            .OrderByDescending(x => x.UpdatedUtc)
+            .Select(x => (DateTime?)x.UpdatedUtc)
+            .FirstOrDefaultAsync(cancellationToken);
+        var lastBalanceUtc = await _dbContext.FinanceBalances
+            .Where(x => x.CompanyId == companyId)
+            .OrderByDescending(x => x.CreatedUtc)
+            .Select(x => (DateTime?)x.CreatedUtc)
+            .FirstOrDefaultAsync(cancellationToken);
+        var lastGeneratedUtc = new[] { lastTransactionUtc, lastInvoiceUtc, lastBalanceUtc }
+            .Where(x => x.HasValue)
+            .Select(x => x!.Value)
+            .DefaultIfEmpty(DateTime.UtcNow)
+            .Max();
+
+        return new FinanceSandboxDatasetGenerationResponse
+        {
+            ProfileName = "Tenant sandbox dataset",
+            LastGeneratedUtc = lastGeneratedUtc,
+            CoverageSummary = $"{transactionCount} transactions, {invoiceCount} invoices, and {balanceCount} balances are available in the active sandbox dataset.",
+            AvailableProfiles =
+            [
+                "Tenant sandbox dataset",
+                "Quarter-close reconciliation pack",
+                "Stress cashflow replay"
+            ]
+        };
+    }
