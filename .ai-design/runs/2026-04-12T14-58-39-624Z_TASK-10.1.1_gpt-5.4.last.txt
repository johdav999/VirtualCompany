+            command.Title,
+            command.Description,
+            priority,

+            HumanActorType,
+            membership.UserId,
+            command.InputPayload,
+            command.ParentTaskId,
+            command.WorkflowInstanceId);
+        task.SetDueDate(command.DueAt);
+
+        _dbContext.WorkTasks.Add(task);
+        await _dbContext.SaveChangesAsync(cancellationToken);
+
+        return await ToDetailDtoAsync(companyId, task.Id, cancellationToken);
+    }

+public sealed class TasksController : ControllerBase
+{
+    private readonly ICompanyTaskService _taskService;
+
+    public TasksController(ICompanyTaskService taskService)
+    {
+        _taskService = taskService;
+    }
+
+    [HttpPost]
+    public async Task<ActionResult<TaskDetailDto>> CreateAsync(
+        Guid companyId,
+        [FromBody] CreateTaskCommand command,
+        CancellationToken cancellationToken)
+    {
+        try
+        {
+            var task = await _taskService.CreateTaskAsync(companyId, command, cancellationToken);
+            return Created($"/api/companies/{companyId}/tasks/{task.Id}", task);
+        }
+        catch (TaskValidationException ex)
+        {
+            return ValidationProblem(ex.Errors);
+        }
+        catch (AgentAssignmentValidationException ex)
+        {
+            return ValidationProblem(ex.Errors);
+        }
+        catch (KeyNotFoundException)
+        {
+            return NotFound();
+        }
+        catch (UnauthorizedAccessException)
+        {
+            return Forbid();
+        }
+    }

+        ValidationProblem(new ValidationProblemDetails(new Dictionary<string, string[]>(errors, StringComparer.OrdinalIgnoreCase))
+        {
+            Title = "Validation failed",
+            Status = StatusCodes.Status400BadRequest
+        });
