+    private static string FormatBriefingReferences(IReadOnlyList<BriefingSourceReferenceViewModel> references)
+    {
+        var groups = references
+            .Where(reference => !string.IsNullOrWhiteSpace(reference.Label))
+            .GroupBy(reference => ReferenceGroupLabel(reference.EntityType))
+            .OrderBy(group => group.Key)
+            .Select(group =>
+            {
+                var labels = group
+                    .Take(3)
+                    .Select(ReferenceLabel);
+                return $"Related {group.Key}: {string.Join(", ", labels)}";
+            })
+            .ToList();
+
+        return groups.Count == 0 ? "Related items available." : string.Join(" | ", groups);
+    }
