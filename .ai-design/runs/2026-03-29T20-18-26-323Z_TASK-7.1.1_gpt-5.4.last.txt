+using System.Security.Claims;
+using System.Text.Encodings.Web;
+using Microsoft.AspNetCore.Authentication;
+using Microsoft.Extensions.Options;
+
+namespace VirtualCompany.Api.Auth;
+
+public sealed class HeaderAuthenticationHandler : AuthenticationHandler<AuthenticationSchemeOptions>
+{
+    public HeaderAuthenticationHandler(
+        IOptionsMonitor<AuthenticationSchemeOptions> options,
+        ILoggerFactory logger,
+        UrlEncoder encoder)
+        : base(options, logger, encoder)
+    {
+    }
+
+    protected override Task<AuthenticateResult> HandleAuthenticateAsync()
+    {
+        var provider = Request.Headers[HeaderAuthenticationDefaults.ProviderHeader].ToString();
+        var subject = Request.Headers[HeaderAuthenticationDefaults.SubjectHeader].ToString();
+
+        if (string.IsNullOrWhiteSpace(provider) || string.IsNullOrWhiteSpace(subject))
+        {
+            return Task.FromResult(AuthenticateResult.NoResult());
+        }
+
+        var claims = new List<Claim>
+        {
+            new("sub", subject),
+            new("identity_provider", provider),
+            new("iss", HeaderAuthenticationDefaults.Issuer)
+        };
+
+        var email = Request.Headers[HeaderAuthenticationDefaults.EmailHeader].ToString();
+        if (!string.IsNullOrWhiteSpace(email))
+        {
+            claims.Add(new Claim(ClaimTypes.Email, email));
+        }
+
+        var name = Request.Headers[HeaderAuthenticationDefaults.NameHeader].ToString();
+        if (!string.IsNullOrWhiteSpace(name))
+        {
+            claims.Add(new Claim(ClaimTypes.Name, name));
+        }
+
+        var identity = new ClaimsIdentity(claims, Scheme.Name);
+        var principal = new ClaimsPrincipal(identity);
+        var ticket = new AuthenticationTicket(principal, Scheme.Name);
+
+        return Task.FromResult(AuthenticateResult.Success(ticket));
+    }
+}

+using VirtualCompany.Domain.Companies;
+using VirtualCompany.Domain.Identity;
+
+namespace VirtualCompany.Application.Auth;
+
+public sealed class CurrentUserMembershipService : ICurrentUserMembershipService
+{
+    private readonly IAuthenticatedPrincipalAccessor _authenticatedPrincipalAccessor;
+    private readonly IUserRepository _userRepository;
+    private readonly ICompanyMembershipRepository _companyMembershipRepository;
+
+    public CurrentUserMembershipService(
+        IAuthenticatedPrincipalAccessor authenticatedPrincipalAccessor,
+        IUserRepository userRepository,
+        ICompanyMembershipRepository companyMembershipRepository)
+    {
+        _authenticatedPrincipalAccessor = authenticatedPrincipalAccessor;
+        _userRepository = userRepository;
+        _companyMembershipRepository = companyMembershipRepository;
+    }
+
+    public async Task<CurrentUserResult> GetCurrentUserAsync(CancellationToken cancellationToken)
+    {
+        var user = await ResolveUserAsync(cancellationToken);
+        return await BuildResultAsync(user, cancellationToken);
+    }
+
+    public async Task<CurrentUserResult> GetMembershipsAsync(CancellationToken cancellationToken)
+    {
+        var user = await ResolveUserAsync(cancellationToken);
+        return await BuildResultAsync(user, cancellationToken);
+    }
+
+    public async Task<CurrentUserResult> SelectActiveCompanyAsync(Guid companyId, CancellationToken cancellationToken)
+    {
+        var user = await ResolveUserAsync(cancellationToken);
+        var memberships = await _companyMembershipRepository.GetByUserIdAsync(user.Id, cancellationToken);
+        var selectedMembership = memberships.SingleOrDefault(x => x.CompanyId == companyId && x.Status == MembershipStatus.Active);
+
+        if (selectedMembership is null)
+        {
+            throw new InvalidActiveCompanySelectionException("The current user does not have an active membership for the selected company.");
+        }
+
+        user.SetActiveCompany(companyId, DateTimeOffset.UtcNow);
+        await _userRepository.UpdateAsync(user, cancellationToken);
+
+        return await BuildResultAsync(user, memberships, cancellationToken);
+    }
+
+    private async Task<User> ResolveUserAsync(CancellationToken cancellationToken)
+    {
+        var principal = _authenticatedPrincipalAccessor.GetRequiredPrincipal();
+        var user = await _userRepository.GetByAuthAsync(principal.Provider, principal.Subject, cancellationToken);
+
+        if (user is not null)
+        {
+            if (user.RefreshIdentity(principal.Email, principal.DisplayName, DateTimeOffset.UtcNow))
+            {
+                await _userRepository.UpdateAsync(user, cancellationToken);
+            }
+
+            return user;
+        }
+
+        if (!string.IsNullOrWhiteSpace(principal.Email))
+        {
+            var emailCandidates = await _userRepository.GetByNormalizedEmailAsync(User.NormalizeEmail(principal.Email)!, cancellationToken);
+            if (emailCandidates.Count == 1 && !emailCandidates[0].HasLinkedAuth)
+            {
+                user = emailCandidates[0];
+                user.LinkAuthProvider(principal.Provider, principal.Subject, principal.Email, principal.DisplayName, DateTimeOffset.UtcNow);
+                await _userRepository.UpdateAsync(user, cancellationToken);
+                return user;
+            }
+        }
+
+        user = User.Create(principal.Provider, principal.Subject, principal.Email, principal.DisplayName, DateTimeOffset.UtcNow);
+        await _userRepository.AddAsync(user, cancellationToken);
+        return user;
+    }
+
+    private async Task<CurrentUserResult> BuildResultAsync(User user, CancellationToken cancellationToken)
+    {
+        var memberships = await _companyMembershipRepository.GetByUserIdAsync(user.Id, cancellationToken);
+        return await BuildResultAsync(user, memberships, cancellationToken);
+    }
+
+    private async Task<CurrentUserResult> BuildResultAsync(
+        User user,
+        IReadOnlyList<CompanyMembership> memberships,
+        CancellationToken cancellationToken)
+    {
+        var activeMemberships = memberships
+            .Where(x => x.Status == MembershipStatus.Active)
+            .OrderBy(x => x.Company.Name, StringComparer.OrdinalIgnoreCase)
+            .ThenBy(x => x.CompanyId)
+            .ToList();
+
+        var selectedMembership = user.ActiveCompanyId.HasValue
+            ? activeMemberships.SingleOrDefault(x => x.CompanyId == user.ActiveCompanyId.Value)
+            : null;
+
+        var now = DateTimeOffset.UtcNow;
+        if (selectedMembership is null && activeMemberships.Count == 1)
+        {
+            selectedMembership = activeMemberships[0];
+            if (user.ActiveCompanyId != selectedMembership.CompanyId)
+            {
+                user.SetActiveCompany(selectedMembership.CompanyId, now);
+                await _userRepository.UpdateAsync(user, cancellationToken);
+            }
+        }
+        else if (selectedMembership is null && user.ActiveCompanyId.HasValue)
+        {
+            user.SetActiveCompany(null, now);
+            await _userRepository.UpdateAsync(user, cancellationToken);
+        }
+
+        var orderedMemberships = memberships
+            .OrderBy(x => x.Company.Name, StringComparer.OrdinalIgnoreCase)
+            .ThenBy(x => x.CompanyId)
+            .Select(x => new CompanyMembershipSummary(
+                x.Id,
+                x.CompanyId,
+                x.Company.Name,
+                x.Role,
+                x.Status,
+                x.Status == MembershipStatus.Active))
+            .ToList();
+
+        return new CurrentUserResult(
+            user.Id,
+            user.AuthProvider,
+            user.AuthSubject,
+            user.Email,
+            user.DisplayName,
+            selectedMembership?.CompanyId,
+            selectedMembership?.Company.Name,
+            selectedMembership?.Role,
+            selectedMembership is null && activeMemberships.Count > 1,
+            orderedMemberships);
+    }
+}

+public sealed class ClaimsPrincipalAuthenticatedPrincipalAccessor : IAuthenticatedPrincipalAccessor
+{
+    private readonly IHttpContextAccessor _httpContextAccessor;
+
+    public ClaimsPrincipalAuthenticatedPrincipalAccessor(IHttpContextAccessor httpContextAccessor)
+    {
+        _httpContextAccessor = httpContextAccessor;
+    }
+
+    public AuthenticatedPrincipal GetRequiredPrincipal()
+    {
+        var principal = _httpContextAccessor.HttpContext?.User;
+        if (principal?.Identity?.IsAuthenticated != true)
+        {
+            throw new InvalidOperationException("An authenticated principal is required.");
+        }
+
+        var provider = principal.FindFirstValue("identity_provider")
+            ?? principal.FindFirstValue("idp")
+            ?? principal.Identity?.AuthenticationType;
+        var subject = principal.FindFirstValue("sub")
+            ?? principal.FindFirstValue(ClaimTypes.NameIdentifier)
+            ?? principal.FindFirstValue("oid");
+        var email = principal.FindFirstValue(ClaimTypes.Email) ?? principal.FindFirstValue("email");
+        var displayName = principal.FindFirstValue(ClaimTypes.Name)
+            ?? principal.FindFirstValue("name")
+            ?? principal.FindFirstValue("preferred_username");
+
+        if (string.IsNullOrWhiteSpace(provider) || string.IsNullOrWhiteSpace(subject))
+        {
+            throw new InvalidOperationException("The authenticated principal is missing required provider or subject claims.");
+        }
+
+        return new AuthenticatedPrincipal(provider, subject, email, displayName);
+    }
