+public static class FortnoxWritePayloadSanitizer
+{
+    private static readonly string[] SensitiveNames =
+    [
+        "access_token",
+        "accessToken",
+        "refresh_token",
+        "refreshToken",
+        "client_secret",
+        "clientSecret",
+        "authorization_code",
+        "authorizationCode",
+        "code",
+        "token",
+        "secret"
+    ];
+
+    public static string CreateSummary<T>(T payload)
+    {
+        if (payload is null)
+        {
+            return "No payload body.";
+        }
+
+        var redacted = Redact(JsonSerializer.SerializeToNode(payload, FortnoxJson.Options));
+        var text = redacted?.ToJsonString(FortnoxJson.Options) ?? "{}";
+        return text.Length <= 500 ? text : string.Concat(text.AsSpan(0, 497), "...");
+    }
+
+    public static string CreatePayloadHash<T>(T payload)
+    {
+        var redacted = payload is null
+            ? "{}"
+            : Redact(JsonSerializer.SerializeToNode(payload, FortnoxJson.Options))?.ToJsonString(FortnoxJson.Options) ?? "{}";
+        var bytes = SHA256.HashData(Encoding.UTF8.GetBytes(redacted));
+        return Convert.ToHexString(bytes).ToLowerInvariant();
+    }
+
+    private static JsonNode? Redact(JsonNode? node)
+    {
+        if (node is JsonObject obj)
+        {
+            foreach (var property in obj.ToList())
+            {
+                obj[property.Key] = IsSensitive(property.Key) ? "*** redacted ***" : Redact(property.Value);
+            }
+        }
+        else if (node is JsonArray array)
+        {
+            for (var index = 0; index < array.Count; index++)
+            {
+                array[index] = Redact(array[index]);
+            }
+        }
+
+        return node;
+    }
+
+    private static bool IsSensitive(string name) =>
+        SensitiveNames.Any(sensitive => name.Contains(sensitive, StringComparison.OrdinalIgnoreCase));

+    public async Task EnsureApprovedAsync(FortnoxWriteApprovalCheck check, CancellationToken cancellationToken)
+    {
+        if (check.ApprovedApprovalId is not Guid approvalId)
+        {
+            var created = await CreateApprovalAsync(check, cancellationToken);
+            throw new FortnoxApprovalRequiredException(
+                created.Id,
+                "Fortnox writes require approval before data is sent to Fortnox.");
+        }
+
+        var approval = await _approvalRequestService.GetAsync(check.CompanyId, approvalId, cancellationToken);
+        if (!string.Equals(approval.Status, "approved", StringComparison.OrdinalIgnoreCase))
+        {
+            throw new FortnoxApprovalRequiredException(
+                approval.Id,
+                "Fortnox write approval is not approved.");
+        }
+
+        await AddApprovedWriteAuditAsync(check, approval, cancellationToken);
+    }

+    private async Task<ApprovalRequestDto> CreateApprovalAsync(
+        FortnoxWriteApprovalCheck check,
+        CancellationToken cancellationToken)
+    {
+        var target = await ResolveTargetCompanyAsync(check.CompanyId, cancellationToken);
+        var context = new Dictionary<string, JsonNode?>
+        {
+            ["provider"] = "Fortnox",
+            ["targetCompany"] = target,
+            ["direction"] = "outbound",
+            ["entityType"] = check.EntityType,
+            ["httpMethod"] = check.HttpMethod,
+            ["path"] = check.Path,
+            ["payloadSummary"] = check.PayloadSummary,
+            ["payloadHash"] = check.PayloadHash
+        };
+
+        return await _approvalRequestService.CreateAsync(
+            check.CompanyId,
+            new CreateApprovalRequestCommand(
+                "fortnox_write",
+                check.WriteRequestId,
+                "human",
+                ResolveRequester(check),
+                ApprovalType,
+                context,
+                RequiredRole: "finance_approver"),
+            cancellationToken);
+    }
+
+    private async Task AddApprovedWriteAuditAsync(
+        FortnoxWriteApprovalCheck check,
+        ApprovalRequestDto approval,
+        CancellationToken cancellationToken)
+    {
+        var alreadyRecorded = await _dbContext.FinanceIntegrationAuditEvents
+            .AsNoTracking()
+            .AnyAsync(x =>
+                x.CompanyId == check.CompanyId &&
+                x.ProviderKey == FinanceIntegrationProviderKeys.Fortnox &&
+                x.EventType == "approved_write" &&
+                x.CorrelationId == approval.Id.ToString("N"),

+        }
